    function fetchDataAndUpdateChart() {
        // Fetch data from PHP script
        fetch('/data') // Update the path to your PHP script
            .then(response => response.json())
            .then(data => {
                if (data.length === 0) {
                    // If no data is returned, create an empty chart
                    createEmptyChart();
                    return;
                }
                // Prepare data for the chart
                const Tests = data.map(entry => entry[0]);
                const Voltage = data.map(entry => entry[1]);
                const pH = data.map(entry => entry[2]);
                const Temp = data.map(entry => entry[3]);
                const AirQuality = data.map(entry => entry[4]);
                
                // Get the existing chart instance
                const existingChart = Chart.getChart('ROV_DATA');
                
                // Check if the chart exists
                if (existingChart) {
                    // Update the existing chart with new data
                    existingChart.data.labels = Tests;
                    existingChart.data.datasets[0].data = Voltage;
                    existingChart.data.datasets[1].data = pH;
                    existingChart.data.datasets[2].data = Temp;
                    existingChart.data.datasets[3].data = AirQuality;
                    existingChart.update(); // Update the chart
                } else {
                    // Create a new chart if it doesn't exist
                    createNewChart(Tests, Voltage, pH, Temp, AirQuality);
                }
            })
            .catch(error => {
                console.error('Error fetching data:', error);
            });
    }
